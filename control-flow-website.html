üßÆ Expressions - Building Blocks of Computation

What are expressions? Expressions are combinations of values, variables, and operators that evaluate to a single value.

üß™ Expression Calculator

JavaScript
// Arithmetic Expressions
let a = 15, b = 4;
console.log(a + b);    // 19
console.log(a - b);    // 11
console.log(a * b);    // 60
console.log(a / b);    // 3.75
console.log(a % b);    // 3
console.log(a ** b);   // 50625
Assignment Operators
let x = 10;
x += 5;    // x = x + 5  ‚Üí x = 15
x -= 3;    // x = x - 3  ‚Üí x = 12
x *= 2;    // x = x * 2  ‚Üí x = 24
x /= 4;    // x = x / 4  ‚Üí x = 6
x++;       // x = x + 1  ‚Üí x = 7
x--;       // x = x - 1  ‚Üí x = 6
Logical Expressions
let age = 20, hasLicense = true;
console.log(age >= 18 && hasLicense);    // true
console.log(age < 16 || !hasLicense);    // false
console.log(!(age < 18));                // true
Python
# Arithmetic Expressions
a, b = 15, 4
print(a + b)     # 19
print(a - b)     # 11
print(a * b)     # 60
print(a / b)     # 3.75
print(a // b)    # 3 (floor division)
print(a % b)     # 3
print(a ** b)    # 50625
Assignment Operators
x = 10
x += 5     # x = x + 5  ‚Üí x = 15
x -= 3     # x = x - 3  ‚Üí x = 12
x *= 2     # x = x * 2  ‚Üí x = 24
x /= 4     # x = x / 4  ‚Üí x = 6.0
x //= 2    # x = x // 2 ‚Üí x = 3.0
x **= 2    # x = x ** 2 ‚Üí x = 9.0
Logical Expressions
age, has_license = 20, True
print(age >= 18 and has_license)    # True
print(age < 16 or not has_license)  # False
print(not (age < 18))               # True

// Basic for loop
for (let i = 1; i <= 5; i++) {
    console.log(`Count: ${i}`);
}

// for...of loop (arrays)
const team = ["mark", "larie", "agatha", "jorome"];
for (const member of team) {
    console.log(`Member: ${member}`);
}
while Loop
// Basic while loop
let countdown = 5;
while (countdown > 0) {
    console.log(`Countdown: ${countdown}`);
    countdown--;
}

// do...while loop
let userInput = 0;
do {
    console.log(`Processing: ${userInput}`);
    userInput++;
} while (userInput < 3);
Loop Control
// break and continue
for (let i = 1; i <= 10; i++) {
    if (i === 3) {
        continue; // Skip 3
    }
    if (i === 7) {
        break; // Stop at 7
    }
    console.log(i);
}
for Loop
# Basic for loop
for i in range(1, 6):
    print(f"Count: {i}")

# for loop with list
team = ["mark", "larie", "agatha", "jorome"]
for member in team:
    print(f"Member: {member}")

# for loop with enumerate
for index, member in enumerate(team):
    print(f"{index}: {member}")
while Loop
# Basic while loop
countdown = 5
while countdown > 0:
    print(f"Countdown: {countdown}")
    countdown -= 1

# while with else
x = 0
while x < 3:
    print(f"x is {x}")
    x += 1
else:
    print("Loop completed normally")
Loop Control
# break and continue
for i in range(1, 11):
    if i == 3:
        continue  # Skip 3
    if i == 7:
        break     # Stop at 7
    print(i)
